=== Shell-Programmierung

.Shell-Skript zum Starten eines Webservers
[source,sh]
----
#!/bin/bash
if [ -n "$1" ];
then
    port=$1;
else
    port=5000;
fi
ip4=$(/sbin/ip -o -4 addr list enp0s25 | awk '{print $4}' | cut -d/ -f1)
echo "running httpd on http://${ip4}:${port}"
ruby -run -e httpd srv -p $port -b $ip4
----
